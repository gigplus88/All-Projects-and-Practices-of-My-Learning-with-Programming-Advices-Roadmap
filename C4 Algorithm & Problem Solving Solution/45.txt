string GetMonthOfYear(enMonthsOfYear MonthOfYear)
{
    switch (MonthOfYear)
    {
    case Jan:
        return "January";
    case Feb:
        return "February";
    case Mar:
        return "March";
    case Apr:
        return "April";
    case Mai:
        return "Mai";
    case Jun:
        return "June";
    case Jul:
        return "July";
    case Aug:
        return "August";
    case Sep:
        return "September";
    case Oct:
        return "October";
    case Nov:
        return "November";
    case Dec:
        return "December";
    default:
        return "Invalid Month";
    }
 
}
