void PrintMatrix1(int arr[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            cout << arr[i][j] << "\t";
            //cout << setw(3) << arr[i][j] << "\t";
        }
        cout << endl;
    }

}

bool IsNumberInMatrix(int Matrix1[3][3], short Rows, short Cols, int NumberToLook)
{

    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            if (Matrix1[i][j] == NumberToLook)
            {

                return true;
            }

        }

    }

    return false;


}


void TheIntersectedNumbersInMatrixs(int Matrix1[3][3], int Matrix2[3][3], short Rows, short Cols)
{
    
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            if (IsNumberInMatrix( Matrix2,  3, 3, Matrix1[i][j]))
            {
                 
                cout << Matrix1[i][j] <<"\t";
            }
            
        }

    }

    
    
}
