select * ,cast(NumberOfVehicls  as float) / cast(TotalVehicles as float) as Perc from
(
select Makes.Make , count(*) as NumberOfVehicls , (select count(*) from VehicleDetails) as TotalVehicles 
 from VehicleDetails join Makes
On VehicleDetails.MakeID = Makes.MakeID
where VehicleDetails.Year between 1950 and 2000
group By Makes.Make) R1
Order By NumberOfVehicls Desc;
