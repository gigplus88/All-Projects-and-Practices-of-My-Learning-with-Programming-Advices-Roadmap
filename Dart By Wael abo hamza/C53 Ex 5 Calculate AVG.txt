void main() {
  List numbers = [3, 1 , 83 , 28 ,12];

  print(GetAverage(numbers));
}
double GetAverage(List nms) {
  var sum = nms.reduce((previousValue, element) => previousValue+ element);
  return (sum / nms.length);
}
