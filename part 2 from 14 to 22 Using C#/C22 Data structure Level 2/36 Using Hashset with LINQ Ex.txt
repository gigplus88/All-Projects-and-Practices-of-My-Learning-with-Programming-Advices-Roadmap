using System;
using System.Collections.Generic;
using System.Linq;


class Program
{
    static void Main()
    {

        // Filter unique elements in array with hashSet
        HashSet<int> Numbers = new HashSet<int> { 1, 2, 2, 3, 4, 4, 4, 5 };

        var HiddenNumbers = Numbers.Where(x => x % 2 != 0);

        foreach (int number in HiddenNumbers)
        {
            Console.WriteLine(number);
        }


        Console.WriteLine("############################");

        var SpecificNumbers = Numbers.Where(x => x > 3 );

        foreach (int number in SpecificNumbers)
        {
            Console.WriteLine(number);
        }

        
        Console.ReadKey();
    }
}
