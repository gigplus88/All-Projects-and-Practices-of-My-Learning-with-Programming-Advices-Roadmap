    <script>
      /*â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
      â”‚ Code Overview
      â”‚ Purpose        â†’ Demonstrate classList: add(), toggle(), contains()
      â”‚ Layout         â†’ One clickable box + status pills + helper buttons
      â”‚ Use Cases      â†’ UI state (active/open), debugging states, DOM events
      â”‚ Key Properties â†’ classList.add / toggle / contains, event listeners
      â””â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€*/

      // âœ… Select the box element from the page
      const box = document.querySelector("#box");

      // âœ… Small UI helpers (to display current state on the page)
      const pillActive = document.querySelector("#pillActive");
      const pillOpen = document.querySelector("#pillOpen");

      // âœ… Buttons (extra controls to test add/remove easily)
      const btnToggleOpen = document.querySelector("#btnToggleOpen");
      const btnRemoveActive = document.querySelector("#btnRemoveActive");
      const btnAddActive = document.querySelector("#btnAddActive");

      // âœ… Add "active" once when the page loads (does NOT remove other classes)
      box.classList.add("active");

      // âœ… Update the pills to reflect current classes (active/open)
      function updateStatus() {
        pillActive.textContent = `active: ${box.classList.contains("active")}`;
        pillOpen.textContent = `open: ${box.classList.contains("open")}`;
      }

      // âœ… Click on the box:
      //    1) toggle "open" class
      //    2) check if it still has "active"
      //    3) log result to console
      box.addEventListener("click", () => {
        // ðŸ”¹ toggle() adds "open" if missing, removes it if present
        box.classList.toggle("open");

        // ðŸ”¹ contains() returns true/false if the class exists
        if (box.classList.contains("active")) {
          console.log("âœ… Box is active");
        } else {
          console.log("âŒ Box is NOT active");
        }

        // ðŸ”¹ Update the UI pills after the click
        updateStatus();
      });

      // âœ… Extra controls for learning (not required, but helpful)

      // Toggle "open" using a button (same effect as clicking the box)
      btnToggleOpen.addEventListener("click", () => {
        box.classList.toggle("open");
        updateStatus();
      });

      // Remove "active" to see contains("active") become false
      btnRemoveActive.addEventListener("click", () => {
        box.classList.remove("active");
        updateStatus();
      });

      // Add "active" back again
      btnAddActive.addEventListener("click", () => {
        box.classList.add("active");
        updateStatus();
      });

      // âœ… Initial status display
      updateStatus();
    </script>
