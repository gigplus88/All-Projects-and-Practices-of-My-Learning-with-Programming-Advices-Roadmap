
using System;
using System.Collections.Generic;
using System.Net.Http;
using System.Net.Http.Json;
using System.Text.Json.Serialization;
using System.Text;
using System.Threading.Tasks;
//using Newtonsoft.Json;

namespace StudentApiClient
{
    class Program
    {
        static readonly HttpClient httpClient = new HttpClient();

        static async Task Main(string[] args)
        {
            httpClient.BaseAddress = new Uri("http://localhost:5039/api/Students/AverageGrade"); // Set this to the correct URI for your API

            await GetAvgStudents();

        }

        static async Task GetAvgStudents()
        {
            try
            {
                Console.WriteLine("\n_____________________________");
                Console.WriteLine("\nFetching AVG students...\n");
                var studentsAVG = await httpClient.GetFromJsonAsync<double>("AverageGrade");
                
                if (studentsAVG != 0)
                {
                    Console.WriteLine($"Students AVG is : {studentsAVG}");
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"An error occurred: {ex.Message}");
            }
        }



    }

}
