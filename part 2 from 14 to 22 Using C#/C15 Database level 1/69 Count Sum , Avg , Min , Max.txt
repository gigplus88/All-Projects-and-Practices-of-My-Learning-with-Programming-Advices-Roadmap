


Select TotallCount = Count(MonthlySalary) ,
       TotallSum = Sum(MonthlySalary),
	   Average = Avg(MonthlySalary),
	   MinSalary = Min(MonthlySalary),
	   MaxSalary = Max(MonthlySalary)

	 from Employees;
