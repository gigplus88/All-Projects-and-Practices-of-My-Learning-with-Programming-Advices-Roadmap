#include <iostream>
#include <cmath>
#include <string>
#include <iomanip>
#include <vector>

using namespace std;


void Swap(int *n1, int *n2)
{
    int temp;

    temp = *n1;
    *n1 = *n2;
    *n2 = temp;

}

int main()
{
    int a = 3 , b=7;

    
    cout << " a before swaping is:" << a << endl;
    cout << " b before swaping is:" << b << endl;

    Swap( &a, &b);


    cout << " a after swaping is:" << a << endl;
    cout << " b after swaping is:" << b << endl;
   
   

    return 0;
}
