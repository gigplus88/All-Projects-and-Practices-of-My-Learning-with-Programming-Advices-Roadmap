  <style>
    /* Global Styles */
    *, *::before, *::after {
      box-sizing: border-box;
    }
    body {
      margin: 0;
      font-family: Arial, sans-serif;
      line-height: 1.4;
      background: #f9f9f9;
      color: #333;
    }
    header {
      background: #4CAF50;
      color: white;
      padding: 1rem;
      text-align: center;
    }
    section {
      margin: 2rem auto;
      padding: 1rem;
      background: white;
      border-radius: 8px;
      max-width: 90%;
      box-shadow: 0 2px 5px rgba(0,0,0,0.1);
    }
    h2 {
      margin-top: 0;
    }

    /* 1. Pixels */
    .pixels {
      padding: 20px;        /* 20px padding */
      border: 2px solid #2196F3; /* 2px border */
    }

    /* 2. Inches & Centimeters */
    .inches-cm {
      width: 4in;           /* 4 inches wide */
      padding: 1cm;         /* 1 cm padding */
      border: 1px dashed #FF5722;
      margin: 1rem auto;
    }

    /* 3. Percentages */
    .percent {
      max-width: 50%;           /* 80% of parent width */
      margin: 0 auto;
      padding: 2%;
      background: #E8F5E9;
      border: 1% solid #4CAF50;
      text-align: center;
    }

    /* 4. em (relative to parent font-size) */
    .em-unit {
      font-size: 1.25em;    /* 1.25 × inherited font-size */
      padding: 1em;         /* 1 × inherited font-size */
      background: #FFF3E0;
      border: 0.1em solid #FB8C00;
    }

    /* 5. rem (relative to root font-size) */
    html { font-size: 16px; }
    .rem-unit {
      font-size: 1.5rem;    /* 1.5 × 16px = 24px */
      padding: 1rem;
      background: #E3F2FD;
      border: 0.25rem solid #1976D2;
    }

    /* 6. Viewport Units */
    .viewport-hero {
      width: 100vw;
      height: 30vh;
      background: #673AB7;
      color: white;
      display: flex;
      align-items: center;
      justify-content: center;
      font-size: 2rem;
      margin-bottom: 2rem;
    }

    /* 7. ch (character width) */
    .ch-unit {
      width: 60ch;          /* 40 characters wide */
      padding: 6ch;
      background: #FCE4EC;
      border: 1px solid #D81B60;
      margin: 0 auto;
    }

    /* 8. ex (x-height) */
    .ex-unit {
      font-size: 18px;
      margin-top: 2ex;      /* 2 × x-height */
      padding: 1ex;         /* 1 × x-height */
      background: #EDE7F6;
      border: 1px solid #5E35B1;
      line-height: 5ex;
    }

    footer {
      text-align: center;
      padding: 4vmin;
      background: #333;
      color: white;
    }
  </style>
</head>
