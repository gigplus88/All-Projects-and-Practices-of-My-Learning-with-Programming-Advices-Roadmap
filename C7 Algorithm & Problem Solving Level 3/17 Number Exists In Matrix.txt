#include <iostream>
#include <string>
#include <iomanip>

using namespace std;



void PrintMatrix1(int arr[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            cout << arr[i][j];
            //cout << setw(3) << arr[i][j] << "\t";
        }
        cout << endl;
    }

}

 bool IsNumberInMatrix(int Matrix1[3][3], short Rows, short Cols, short NumberToLook)
{
    
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            if (Matrix1[i][j] == NumberToLook)
            {
                
                return true;
            }
            
        }

    }

    return false;
    
    
}





int main()
{
    srand((unsigned)time(NULL));

    int Matrix1[3][3] =
    { {0,2,5},
      {0,4,1},
      {5,0,0}
    };
    short NumberToLook=0;

    cout << "Matrix1 :" << endl;
    PrintMatrix1(Matrix1, 3, 3);

    cout << "Please Enter the Number to look for in matrix? ";
    cin >> NumberToLook;

    if (IsNumberInMatrix( Matrix1, 3, 3, NumberToLook))
    {
        cout << "Yes: It is there" << endl;
    }

    else
    {
        cout << "No: It is not there";
    }

    system("pause>0");


    return 0;
}
