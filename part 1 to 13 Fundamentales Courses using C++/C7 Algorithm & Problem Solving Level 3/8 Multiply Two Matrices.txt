#include <iostream>
#include <string>
#include <iomanip>

using namespace std;

int RandomNumber(int From, int To)
{

    int randNum = rand() % (To - From + 1) + From;


    return randNum;
}

void FillMatrixWithRandumNumbers(int Matrix1[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            Matrix1[i][j] = RandomNumber( 1, 10);

        }

    }

}

void FillMatrix2WithOrderNumbers(int Matrix2[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            Matrix2[i][j] = RandomNumber(1, 10);

        }

    }

}



void PrintMatrix(int arr[3][3] , short Rows, short Cols)
{
    cout << "The following is a 3*3 random matrix: " << endl;
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            printf(" %0*d  ", 2, arr[i][j]);
            //cout << setw(3) << arr[i][j] << "\t";
        }
        cout << endl;
    }

}

void FillOfMultiplyMatrix(int Matrix1[3][3], int Matrix2[3][3], int MatrixResults[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            MatrixResults[i][j] = Matrix1[i][j] * Matrix2[i][j];

        }

    }

}




int main()
{
    srand((unsigned)time(NULL));

    int Matrix1[3][3] , Matrix2[3][3] , MatrixResults[3][3];

    FillMatrixWithOrderNumbers(Matrix1, 3, 3);
    cout << "Matrix1" << endl;
    PrintMatrix(Matrix1, 3, 3);

    FillMatrix2WithOrderNumbers(Matrix2, 3, 3);
    cout << "Matrix2" << endl;
    PrintMatrix(Matrix2, 3, 3);

    FillOfMultiplyMatrix( Matrix1,  Matrix2, MatrixResults,  3,  3);
    cout << "Matrix Results" << endl;
    PrintMatrix(MatrixResults, 3, 3);

    system("pause>0");


    return 0;
}
