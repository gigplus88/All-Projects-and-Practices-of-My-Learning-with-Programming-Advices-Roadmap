using System;
using System.Collections.Generic;
using System.Linq;


class Program
{
    static void Main()
    {
        // Creating and populating a HashSet of strings
        HashSet<int> Set1 = new HashSet<int> { 1 ,  2, 3 , 4 , 5 ,1 ,2  , 7 , 8};
        HashSet<int> Set2 = new HashSet<int> { 3, 4, 5 };


        Set1.ExceptWith(Set2);


        Console.WriteLine("Difference of sets (set1 - set2):");
        foreach (int item in Set1)
        {
            Console.WriteLine(item);
        }

        Console.ReadKey();
    }
}








using System;
using System.Collections.Generic;


class Program
{
    static void Main()
    {
        HashSet<int> set1 = new HashSet<int> { 1, 2, 3 , 9 };
        HashSet<int> set2 = new HashSet<int> { 3, 4, 5 , 8 , 4 };


        // Removing elements of set2 from set1
        set1.ExceptWith(set2);


        Console.WriteLine("Difference of sets (set1 - set2):");
        foreach (int item in set1)
        {
            Console.WriteLine(item);
        }
        Console.ReadKey();

    }
}
