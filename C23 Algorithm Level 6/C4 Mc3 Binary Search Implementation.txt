using System;
using System.Diagnostics;


class LinearSearchExample
{
    // Method to perform binary search
    static int BinarySearch(int[] arr, int x) //Searching index of x
    {
        int Start = 0, End = arr.Length - 1;
        int Trials = 0;

        // My Addition
        if (arr[Start] == x)
            return Start;

        // My Addition
        if (arr[End] == x)
            return End;
        

        while (Start <= End)
        {
            
            int Middle = Start + (End - Start) / 2;
            Trials++;

            Console.WriteLine("Trial = " + Trials);

            // Check if x is present at mid
            if (arr[Middle] == x)
                return Middle;

            // If x greater, ignore left half
            if (x > arr[Middle])
                Start = Middle + 1;

            // If x is smaller, ignore right half
            else
                End = Middle - 1;
        }

        // If we reach here, then element was not present
        return -1;
    }
    static int LinearSearch(int[] arr, int x)
    {
        int n = arr.Length;
        for (int i = 0; i < n; i++)
        {
            // Return the index of the element if found
            if (arr[i] == x)
                return i;
        }
        // Return -1 if the element is not found
        return -1;
    }



    // Driver code
    static void Main(string[] args)
    {
        int n = 1_000_000;

        int[] arr = new int[n];

        //int[] arr = { 22, 25, 37, 41, 45, 46, 49, 51, 55, 58, 70, 80, 82, 90, 95 , 232 , 29839 , 4885 , 12 , 23 , 34 ,45 , 56 ,67 , 78 , 90 }; // Sorted array
        int x = 45; // Element to be searched

        Console.WriteLine("Sorted Array:");
        //foreach (var item in arr)
        //    Console.Write(item + " ");
        Console.WriteLine();


        Stopwatch stopwatch = new Stopwatch();
        stopwatch.Start();

        int result = BinarySearch(arr, x);

        stopwatch.Stop();

        Console.WriteLine($"Binary search time has {stopwatch.Elapsed} .");




        Stopwatch stopwatch2 = new Stopwatch();
        stopwatch2.Start();

        int result2 = LinearSearch(arr, x);

        stopwatch2.Stop();
        Console.WriteLine($"Linear search time has {stopwatch2.Elapsed} .");



        if (result == -1)
            Console.WriteLine("Element not found in the array.");
        else
            Console.WriteLine("Element found at index: " + result);

        Console.ReadKey();
    }
}
