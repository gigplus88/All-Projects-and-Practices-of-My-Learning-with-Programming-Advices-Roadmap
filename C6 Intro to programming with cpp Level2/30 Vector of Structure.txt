#include <iostream>
#include <cmath>
#include <string>
#include <iomanip>
#include <vector>

using namespace std;
struct stEmployee 
{
    string FirstName , LastName;
    short Salary=0;
};
void ReadEmployees(vector <stEmployee> &vEmployee)
{
    stEmployee Employee;

    

    char UserAnswer;
    do
    {
        cout << "Please enter your FirstName: ";
        cin >> Employee.FirstName;
        
        cout << "Please enter your LastName: ";
        cin >> Employee.LastName;

        cout << "Please enter your Salary: ";
        cin >> Employee.Salary;
        
        vEmployee.push_back(Employee);

        cout << "Do you want to add more employees Y/N: ";
        cin >> UserAnswer;

    } while (UserAnswer == 'Y' || UserAnswer == 'y');

}

void PrintVectorEmployees(vector <stEmployee>& vEmployee)
{
    cout << "Employee Vectors: " <<endl << endl;

    for (stEmployee& Employee : vEmployee)
    {
        cout << "FirstName: "<< Employee.FirstName <<endl;
        cout << "LastName: " << Employee.LastName << endl;
        cout << "Salary: " << Employee.Salary << endl;
        cout << endl;

    }


}

int main()
{
    vector <stEmployee> vEmployee;

    ReadEmployees(vEmployee);
    PrintVectorEmployees(vEmployee);

    return 0;
}
