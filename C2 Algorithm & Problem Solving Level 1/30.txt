#include <iostream>
#include <cmath>
#include <string>
using namespace std;


void ReadNumber(int& Num)
{
    cout << "Enter your number please\n";
    cin >> Num;
}

void PrintNumber(int& i, int& Num, int& F)
{
    if (Num<0)
    {
        cout << "Factorial must be positive number";
    }
    else
    {
        F = 1;

        for (int i = Num; i >= 1; i--)
        {

            F = F * i;


        }
        cout << F;
    }
   
    
}


int main()
{
    int Num = 0, i = 0, F = 0;

    ReadNumber(Num);
    PrintNumber(i, Num, F);

}
