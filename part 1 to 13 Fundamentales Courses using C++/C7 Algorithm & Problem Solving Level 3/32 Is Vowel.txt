bool ISVowel(  char letter, char arrVowel[5])
{

    for (short i = 0; i < 5; i++)
    {
        if (toupper(letter) == toupper(arrVowel[i]))
        {
            return true;
        }
    }
    return false;
}

bool IsVowel(char letter)
{

    letter = tolower(letter);

    return((letter == 'a') || (letter == 'o') || (letter == 'u') || (letter == 'i') || (letter == 'e'));
}
