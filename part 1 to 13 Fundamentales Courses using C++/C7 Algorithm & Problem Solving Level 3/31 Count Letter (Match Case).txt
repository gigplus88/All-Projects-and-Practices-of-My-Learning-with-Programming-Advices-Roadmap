short CounterOfLetterCase( string S1 , char letter, bool MatchCase = true) 
{
    short Counter = 0 ;

    for (short i = 0; i < S1.length(); i++)
    {
        if (MatchCase)
        {
            if (S1[i] == letter)
            Counter++;            
        }
       
        else
        {
            if (toupper(S1[i]) == toupper(letter))
            {
                Counter++;
            }
        }
    }
    return Counter;
}

char InvertLetterCase(char  Character)
{
    return isupper(Character) ? tolower(Character) : toupper(Character);
}
