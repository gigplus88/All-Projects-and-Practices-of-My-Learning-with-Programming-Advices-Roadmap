string GetDayOfWeek(enDayOfWeeks DayOfWeeks)
{
    switch (DayOfWeeks)
    {
    case Sat:
        return "Saturday";
    case enDayOfWeeks::Sun:
        return "Sunday";
    case enDayOfWeeks::Mon:
        return "Monday";
    case enDayOfWeeks::Tue:
        return "Tuesday";
    case enDayOfWeeks::Wed:
        return "Wednesday";
    case enDayOfWeeks::Thu:
        return "Thuersday";
    case enDayOfWeeks::Fri:
        return "Friday";
    default:
        return "Not a Valide Day";
    }
 
}
