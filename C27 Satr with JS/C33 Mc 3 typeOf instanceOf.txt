    <script>
      // ========================================
      // PART 1 â€” typeof
      // ========================================
      let text = "Hello";
      let number = 42;
      let isActive = true;
      let person = { name: "Sara" };
      let nums = [1, 2, 3];
      let greet = function () {
        return "Hi";
      };

      console.log("=== typeof Examples ===");
      console.log("typeof text      ->", typeof text);
      console.log("typeof number    ->", typeof number);
      console.log("typeof isActive  ->", typeof isActive);
      console.log("typeof person    ->", typeof person);
      console.log("typeof nums      ->", typeof nums); // object (arrays are objects)
      console.log("typeof greet     ->", typeof greet); // function

      // ========================================
      // PART 2 â€” instanceof
      // ========================================
      let today = new Date();

      console.log("\n=== instanceof Examples ===");
      console.log("today instanceof Date   ->", today instanceof Date); // true
      console.log("nums instanceof Array   ->", nums instanceof Array); // true
      console.log("nums instanceof Object  ->", nums instanceof Object); // true

      // ========================================
      // PART 3 â€” Equality Rules
      // ========================================

      console.log("\n=== Equality Rules (== vs ===) ===");

      // Loose vs strict equality
      console.log("5 == '5'        ->", 5 == "5");
      console.log("5 === '5'       ->", 5 === "5");
      console.log("0 == false      ->", 0 == false);
      console.log("0 === false     ->", 0 === false);
      console.log("null == undefined ->", null == undefined);
      console.log("null === undefined ->", null === undefined);
    </script>
