float ReadPositiveNumber(string Message)
{

    {

}

bool CheckPerfect(int Number)
{
     int sum = 0;
    for (int i = 1; i <= Number - 1; i++)
    {

        if (Number % i == 0)
        {
            sum = sum + i;

        }

    }
    return Number == sum;
    
}




void GetPerfectNumbersFrom1ToNumber(int Number)
{
    
    cout << endl << "Perfect numbers from 1 to " << Number << "is : " << endl;

    for (int i = 1; i <= Number; i++)
    {

        if (CheckPerfect(i))
        {
            cout << i << endl;

        }
    }

}
