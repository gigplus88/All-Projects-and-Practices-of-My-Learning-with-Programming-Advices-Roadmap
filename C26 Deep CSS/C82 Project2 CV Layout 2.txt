    What this showcases

    ‚Ä¢ Theming with CSS custom properties (tokens) and a pure-CSS toggle via :has()
    ‚Ä¢ Sticky sidebar, rounded ‚Äúcards‚Äù, soft shadows, subtle gradients

    /* ================================================================
       üé® THEME TOKENS (LIGHT BY DEFAULT)
       Define all design tokens once on :root. We flip them in dark mode.
    ================================================================= */
    :root {
      /* Surfaces & palette */
      --bg:            #f6f8fb;    /* page background */
      --sidebar:       #0ea5a5;    /* sidebar base (teal) */
      --sidebar-ink:   #ecfeff;    /* light ink used on teal sidebar */
      --card:          #ffffff;    /* card/background in main column */
      --ink:           #0f172a;    /* primary text color */
      --muted:         #6b7280;    /* secondary/quiet text */
      --accent:        #f97316;    /* brand/accent (orange) */
      --line:          #e5e7eb;    /* hairline borders/dividers */

      /* Shape & depth */
      --radius:        18px;                     /* global rounding */
      --ring:          0 10px 25px rgba(2,6,23,.08); /* soft shadow */

      /* Spacing scale */
      --space-1: 6px;  --space-2: 10px; --space-3: 14px;
      --space-4: 20px; --space-5: 28px; --space-6: 40px;

      /* Layout caps */

      --maxw:          1300px;     /* page max width */
    }

    /* ================================================================
       üåó DARK THEME VIA PURE CSS
       When the checkbox #themeSwitch is checked, :has() matches on :root
       and we override tokens. No JavaScript required.
    ================================================================= */

      --bg:          #0b1220;
      --sidebar:     #075e5e;
      --sidebar-ink: #c9f3f5;
      --card:        #0f172a;
      --ink:         #e5e7eb;
      --muted:       #9aa3b2;
      --accent:      #fb923c;
      --line:        #1f2a44;
      --ring:        none; /* avoid glow on dark */
    }

    /* ================================================================
       üåì SYSTEM DARK FALLBACK
       If the custom toggle isn‚Äôt present/used, follow OS theme instead.
    ================================================================= */
    @media (prefers-color-scheme: dark) {
      :root:not(:has(#themeSwitch)) {
        --bg:          #0b1220;
        --sidebar:     #075e5e;
        --sidebar-ink: #c9f3f5;

        --ink:         #e5e7eb;
