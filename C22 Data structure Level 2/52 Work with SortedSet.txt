using System;
using System.Collections.Generic;
using System.Linq;



class Program
{
    static void Main()
    {
        SortedSet<int> sortedSet = new SortedSet<int> { 1 , 2,3 ,4};

        Console.WriteLine(string.Join(",", sortedSet));


        foreach (var item in sortedSet)
        {
            Console.WriteLine(item );
        }

        int target = 3;
        if (sortedSet.Contains(target))
        {
            Console.WriteLine($"Element {target} exists in the SortedSet.");
        }

        target = 2;
        if (sortedSet.Contains(target))
        {
            sortedSet.Remove(target);
            Console.WriteLine($"Element {target} is Removed in the SortedSet.");

        }

        Console.WriteLine(string.Join(",", sortedSet));

        sortedSet.Clear();


    }
}
