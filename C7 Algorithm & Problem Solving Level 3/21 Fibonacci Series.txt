#include <iostream>
#include <string>
#include <iomanip>

using namespace std;






void FibonnaciSeries(short HowNumbers)
{
    int Prev1 = 1, Prev2 = 0, NewPrev=0;
    
    cout << Prev1 <<" ";

    for (int i = 1; i <= HowNumbers-1; i++)
    {
        NewPrev = Prev1 + Prev2;
        cout <<  NewPrev <<" ";
        
        Prev2 = Prev1;
        Prev1 = NewPrev;

    }

}





int main()
{
    

    FibonnaciSeries(10);


    return 0;
}
