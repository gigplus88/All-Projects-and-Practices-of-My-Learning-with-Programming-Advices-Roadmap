short CountEachWordInString(string  S1)
{
    string Word, delim = " ";
    short pos = 0 , Counter=0;

    while ((pos = S1.find(delim)) != std::string::npos )
    {
        Word = S1.substr(0, pos);

        if (Word != "")
        {
            Counter++;
            
        }

        S1.erase(0, pos + delim.length());
    }

    if (S1 != "")
    {
        
        Counter++;
    }

    return Counter;


}
