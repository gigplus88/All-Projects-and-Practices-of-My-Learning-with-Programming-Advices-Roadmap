#include <iostream>
#include <cmath>
#include <string>
#include <iomanip>
#include <vector>

using namespace std;



int main()
{
    vector <int> numbers{ 1,2,3,5 };

	cout << "With constant:";
	for (const int &i: numbers )
	{
		cout  << i << " " ;

	}
	cout << endl;

	cout << "Change items with for: ";

	for (int& i : numbers)
	{
		i = 20;
		cout << i << " ";
	}
	cout << endl;


	cout << "Change With indexing: ";

	numbers[0] = 12;
	numbers[1] = 19;
	numbers[3] = 23;

	for (int& i : numbers)
	{
		cout << i << " ";
	}

    return 0;
}
