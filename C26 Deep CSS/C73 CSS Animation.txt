  Key Properties:
    ‚Ä¢ @keyframes ‚Äì define animation stages
    ‚Ä¢ animation-name ‚Äì links element to @keyframes
    ‚Ä¢ animation-duration ‚Äì sets cycle length
    ‚Ä¢ animation-timing-function ‚Äì controls easing
    ‚Ä¢ animation-iteration-count ‚Äì repeat behavior
    ‚Ä¢ animation-direction + fill-mode ‚Äì control playback direction and final state

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>CSS Animations Combined Demo</title>
  <style>
    /* Reset & Base Styles */
    * { box-sizing: border-box; margin: 0; padding: 0; }
    body {
      font-family: Arial, sans-serif;
      background: #fafafa;
      color: #333;
      padding: 2rem;
      max-width: 600px;
      margin: auto;
      line-height: 1.4;
    }
    h1, h2 {
      text-align: center;
      margin-bottom: 1rem;
    }
    section {
      margin-bottom: 3rem;
    }

    /* 1) Bouncing Ball */
    .ball {
      width: 50px; height: 50px;
      background: #3498db;
      border-radius: 50%;
      margin: 1rem auto;
      /* Bounce animation: moves up 100px and back */
     /* name | duration | timing-function | delay | iteration-count | direction | fill-mode | play-state */
      /* animation: bounce 1.5s ease-in-out infinite alternate both; */

        animation-name: bounce;                    /* Links to the @keyframes rule named ‚Äúbounce‚Äù */
        animation-duration: 4s;                  /* One full cycle takes 1.5 seconds */
        animation-timing-function: ease-in-out;    /* Speeds up in the middle, slows at start/end */
        animation-delay: 0s;                       /* No delay‚Äîanimation begins immediately */
        animation-iteration-count:infinite;       /* Repeats forever */
        animation-direction:normal;            /* Reverses direction every other cycle */
        animation-fill-mode:both;                 /* Keeps both the start (0%) and end (100%) styles applied */
        animation-play-state:running;             /* Animation is currently running (not paused) */
    }
    @keyframes bounce {
      0%   { transform: translateX(0px); }
      25%   { transform: translateX(-200px); }
      50% { transform: translateX(0px); }
      75% { transform: translateX(200px); }
      100%   { transform: translateX(0px); }
    }

    /* 2) Fading Text */
    .fade-text {
      font-size: 2rem;
      text-align: center;
      margin: 1rem auto;
      /* Fade animation: opacity 0 ‚Üî 1 */
      animation: fade 3s ease-in-out infinite alternate both;
    }
    @keyframes fade {
      0%   { opacity: 0; }
      100% { opacity: 1; }
    }

    /* 3) Spinning Loader */
    .spinner {
      width: 40px; height: 40px;
      margin: 1rem auto;
      border: 5px solid #eee;
      border-top-color: #3498db;
      border-radius: 50%;
      /* Spin animation: continuous 360¬∞ rotation */
      animation: spin 1s ease-in-out infinite;
    }
    @keyframes spin {
      0%   { transform: rotate(0deg); }
      100% { transform: rotate(360deg); }
    }


    /* 4) Pulsing Button */
    .pulse-btn-container {
      text-align: center;
    }
    .pulse-btn {
      background: #3498db;
      color: #fff;
      border: none;
      padding: 0.75rem 1.5rem;
      font-size: 1rem;
      border-radius: 4px;
      cursor: pointer;
      animation: pulse 2s infinite;
      outline: none;
    }
    @keyframes pulse {
      0%   { transform: scale(1); }
      50%  { transform: scale(1.1); }
      75%  { transform: scale(0.9); }
      100% { transform: scale(1); }
    }

  </style>
</head>
