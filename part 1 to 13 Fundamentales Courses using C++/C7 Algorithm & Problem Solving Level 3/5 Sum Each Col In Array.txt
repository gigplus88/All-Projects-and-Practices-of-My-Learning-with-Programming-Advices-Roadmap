#include <iostream>
#include <string>
#include <iomanip>

using namespace std;

int RandomNumber(int From, int To)
{

    int randNum = rand() % (To - From + 1) + From;


    return randNum;
}

void FillMatrix(int arr[3][3], short Rows, short Cols)
{
    int sum1 = 0, sum2 = 0, sum3 = 0;
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            arr[i][j] = RandomNumber(1, 100);

        }

    }

}

void PrintMatrix(int arr[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            cout << setw(3) << arr[i][j] << "\t";
        }
        cout << endl;
    }

}

int PrintSumOfOneCol(int arr[3][3], short Rows, short Cols)
{

    int sum = 0;
    for (int j = 0; j < Rows; j++)
    {
        sum += arr[j][Cols];

    }

    return sum;
}

void FillArrayOfSum(int arr[3][3], int arrSum[3], short Rows, short Cols )
{
    for (int i = 0; i < Cols; i++)
    {

        arrSum[i] = PrintSumOfOneCol(arr, Rows, i);

    }

}

void PrintSumOfEachCol(int arrSum[3], short Cols)
{
    for (int i = 0; i < Cols; i++)
    {
        cout << "Sum of the col " << i + 1 << " is:" <<
            arrSum[i] << endl;

    }

}


int main()
{
    srand((unsigned)time(NULL));

    int arr[3][3] , arrSum[3];

    FillMatrix(arr, 3, 6);
    PrintMatrix(arr, 3, 6);

    FillArrayOfSum(arr, arrSum, 3, 6),
    PrintSumOfEachCol(arrSum, 6);

    system("pause>0");


    return 0;
}
