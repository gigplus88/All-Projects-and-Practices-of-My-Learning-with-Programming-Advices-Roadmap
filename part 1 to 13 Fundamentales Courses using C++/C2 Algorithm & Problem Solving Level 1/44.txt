enum enWeekDay { Sun = 1, Mon = 2, Tue = 3, Wed = 4, Thu = 5, Fri = 6, Sat = 7 };

{
    cout << "**********************************" << endl;

    cout << "1: Sunday" << endl;
    cout << "2: Monday" << endl;
    cout << "3: Tuesday" << endl;
    cout << "4: Wednesday" << endl;
    cout << "5: Thursday" << endl;
    cout << "6: Friday" << endl;
    cout << "7: Saturday" << endl;
    cout << "**********************************" << endl;
    cout << "Please enter the number of Day?" << endl;

}

enWeekDay ReadNumOfDay()
{
    enWeekDay WeekDay;
    WeekDay = enWeekDay::Sun;

    short int wd;
    cin >> wd;
    return  (enWeekDay)wd;
}

string GetWeekDayName(enWeekDay WeekDay)
{

    switch (WeekDay)
    {
    case enWeekDay::Sun:
        return "Sunday";
        break;
    case enWeekDay::Mon:
        return "Monday";
        break;
    case enWeekDay::Tue:
        return "Tuesday";
        break;
    case enWeekDay::Wed:
        return "Wednesday";
        break;
    case enWeekDay::Thu:
        return "Thuersday";
        break;
    case enWeekDay::Fri:
        return "Friaday";
        break;
    case enWeekDay::Sat:
        return "Saturday";
        break;

    }
}
int main()
{
    


    ShowDay();

    return 0;

}

void ForPrintSorry()
{

int main()
{

    int Month;

    ReadDay(Month);

    switch (Month)
    {
    case 1:
        PrintSunday();
        break;
    case 2:
        PrintMonday();
        break;
    case 3:
        PrintTuesday();
        break;
    case 4:
        PrintWednesday();
        break;
    case 5:
        PrintThursday();

    case 6:
