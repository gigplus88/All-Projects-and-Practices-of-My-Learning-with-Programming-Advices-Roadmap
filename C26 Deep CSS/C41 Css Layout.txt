    /* —— Reset & Base Styles —— */

      margin: 0;
      padding: 0;
      box-sizing: border-box;
    }
    body {
      font-family: Arial, sans-serif;
      line-height: 1.5;
      padding: 1rem;
    }
    h1, h2 {
      margin-bottom: 1rem;

    .section {
      margin-bottom: 2rem;
      padding: 1rem;
      border: 1px solid #ccc;
      border-radius: 4px;
    }

    /* —— 1) Shift It —— */
    .shift-box {
      position: relative;  /* remains in flow */
      top: 30px;           /* moves down 30px */
      left: 30px;          /* moves right 30px */
      width: 100px;
      height: 100px;
      background: #8fa;
      display: flex;
      align-items: center;
      justify-content: center;
      color: #fff;
      font-weight: bold;
    }

    /* —— 2) Overlay —— */
    .overlay-container {
      position: relative;  /* parent for absolute child */
      background: #eef;
      padding: 1rem;
      height: 150px;
    }
    .overlay-child {
      position: absolute;  /* removed from flow */
      top: 0;              
      right: 0;            
      background: #c88;
      color: #fff;
      padding: 0.5rem;
    }

    /* —— 3) Sticky Header —— */
    .sticky-header {
      position: sticky;
      top: 0;
      background: #333;
      color: #fff;
      padding: 1rem;
      z-index: 10;        /* stay above other content */
    }
    .filler {
      height: 800px;
      background: linear-gradient(to bottom, #fff, #ddd);
      margin-top: 1rem;
    }

    /* —— 4) Fixed Badge —— */

      position: fixed;
      bottom: 20px;
      right: 20px;
      background: #28a;
      color: #fff;
      padding: 0.75rem 1rem;
      border: none;
      border-radius: 4px;
      cursor: pointer;
      box-shadow: 0 2px 6px rgba(0,0,0,0.2);
    }
  </style>
</head>
