<!--
  Code Overview: Top Navigation Menu with Animation
  -------------------------------------------------
  Purpose:
    â€¢ Create a responsive top navigation bar that animates into view on page load.
    â€¢ Add a sliding underline effect on menu item hover.

  Layout:
    â€¢ A fixed <nav> at the top of the page containing an unordered list of links.
    â€¢ Main page content below to demonstrate the fixed header.

  Use Cases:
    â€¢ Modern website headers that draw attention.
    â€¢ Animated underlines for interactive feedback on menu items.

  Key Properties & Techniques:
    â€¢ @keyframes slideDown â€“ nav slides in from above.
    â€¢ animation-name / animation-duration / animation-timing-function.
    â€¢ transition on pseudo-element ::after for underline slide.
    â€¢ fixed positioning for persistent header.
-->

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Animated Top Navigation Menu</title>
  <style>
    /* Reset & Base */
    * { box-sizing: border-box; margin: 0; padding: 0; }
    body {
      font-family: Arial, sans-serif;
      line-height: 1.5;
      background: #f4f4f4;
    }

    /* 1) Slide-down animation for the nav */
    @keyframes slideDown {
      0%   { transform: translateY(-100%); }
      100% { transform: translateY(0); }
    }

    /* 2) Top Navigation Bar */
    nav {
      position: fixed;
      top: 0; left: 0; right: 0;
      background: #2c3e50;
      animation-name: slideDown;
      animation-duration: 0.5s;
      animation-timing-function: ease-out;
      z-index: 1000;
    }
    nav ul {
      display: flex;
      justify-content: center;
      list-style: none;
      padding: 1rem;
    }
    nav li {
      margin: 0 1.5rem;
      position: relative; /* for underline pseudo-element */
    }
    nav a {
      color: #ecf0f1;
      text-decoration: none;
      font-size: 1rem;
      padding: 0.25rem 0;
      transition: color 0.3s ease;
    }
    nav a:hover {
      color: #1abc9c;
    }

    /* 3) Sliding underline effect */
    nav a::after {
      content: "";
      position: absolute;
      left: 0; bottom: -4px;
      width: 0; height: 2px;
      background: #1abc9c;
      transition: width 0.3s ease;
    }
    nav a:hover::after {
      width: 100%;
    }

    /* 4) Page content to demonstrate fixed nav */
    main {
      padding: 6rem 2rem 2rem; /* top padding to clear fixed nav */
    }
    .section {
      max-width: 800px;
      margin: 2rem auto;
      background: #fff;
      padding: 1.5rem;
      border-radius: 4px;
      box-shadow: 0 2px 8px rgba(0,0,0,0.1);
    }
  </style>
</head>
