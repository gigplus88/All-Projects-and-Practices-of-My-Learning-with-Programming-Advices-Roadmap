#include <iostream>
#include <cmath>
#include <string>
#include <iomanip>
#include <vector>

using namespace std;



int main()
{
    int arr[4] = {1,3,7,8};
    int * ptr;
    ptr = arr;

    
    cout << "Adresses of array items" << endl;
    cout << ptr << endl;
    cout << ptr +1  << endl; //&arr[1]
    cout << ptr +2 << endl;
    cout << ptr +3<< endl;


    cout << endl << "Values of array items:"<< endl;

    cout << *ptr << endl;
    cout << *(ptr + 1) << endl; //&arr[1]
    cout << *(ptr + 2 )<< endl;
    cout << *(ptr + 3 )<< endl << endl;

    for (int i = 0; i <= 3; i++)
    {
        cout << *(ptr + i) << endl;
    }

    return 0;
}
