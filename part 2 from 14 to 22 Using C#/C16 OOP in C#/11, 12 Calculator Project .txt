using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Simple.Calculator
{
    class  clsCalculator
    {
        private enum enOperation{ Add=1 , Subtract = 2 , Mutiply = 3 , Division = 4 , Clear = 5};

        enOperation Operation;

        private float _OperationValue;
        private float _Result;

        public void PrintResult()
        {
            Console.Write("Result After ");
            
            switch (Operation)
            {
                case enOperation.Add:
                    Console.Write("Adding");
                    break;
                
                case enOperation.Subtract: 
                    Console.Write("Subtracting");
                    break;

                case enOperation.Mutiply:
                    Console.Write("Mutiplying");
                    break;

                case enOperation.Division:
                    Console.Write("Dividing");
                    break;

                case enOperation.Clear:
                    Console.Write("Clear");
                    break;

            }

            Console.Write(" {0}", _OperationValue);
            Console.WriteLine(" is: {0}", _Result);

        }

        public void Clear()
        {
            Operation = enOperation.Clear;
            _OperationValue =0 ;
            _Result = 0;
        }

        public void ADD(float OperationValue)
        {
            Operation = enOperation.Add;
            _OperationValue = OperationValue;
            _Result += OperationValue;
            
        }

        public void Subtract(float OperationValue)
        {
            Operation = enOperation.Subtract;
            _OperationValue = OperationValue;
            _Result -= OperationValue;

        }

        public void Multiply(float OperationValue)
        {
            Operation = enOperation.Mutiply;
            _OperationValue = OperationValue;
            _Result *= OperationValue;

        }

        public void Divide (float OperationValue)
        {
            Operation = enOperation.Division;

            if (OperationValue == 0)
            {
                _OperationValue = OperationValue;
                _Result /= 1;

            }
            else
            {
                _OperationValue = OperationValue;
                _Result /= OperationValue;

            }

        }

        public float GetFinalResults()
        {
            return _Result;
        }

    }
    internal class Program
    {
        static void Main(string[] args)
        {

            clsCalculator Calculator1 = new clsCalculator();

            Calculator1.Clear();

            Calculator1.ADD(10);
            Calculator1.PrintResult();

            Calculator1.ADD(100);
            Calculator1.PrintResult();

            Calculator1.Subtract(20);
            Calculator1.PrintResult();

            Calculator1.Divide(0);
            Calculator1.PrintResult();

            Calculator1.Divide(2);
            Calculator1.PrintResult();

            Calculator1.Multiply(3);
            Calculator1.PrintResult();

            Calculator1.Clear();
            Calculator1.PrintResult();

            


        }
    }
}
