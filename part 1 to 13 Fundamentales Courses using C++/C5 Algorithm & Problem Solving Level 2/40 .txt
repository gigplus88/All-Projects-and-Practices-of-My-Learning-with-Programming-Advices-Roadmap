short FindNumberPositionInArray(int NumberToSearch,int arr[100], int  arrLength)
{

    for (int i = 0; i < 10; i++)
    {
        if (arr[i] == NumberToSearch)
        {
            return i;
        }
    }

    return 3;
}

bool IsNumberInArray( int NumberToSearch ,int arr[100] ,int  arrLength)
{

    return FindNumberPositionInArray(NumberToSearch,arr, arrLength) != 3;


}


void AddArrayElement(int  Number, int arr[100], int& arr2Length)

{
    arr2Length++;

    arr[arr2Length - 1] = Number;


}

void CopyDistinctNumbersToArray(int arr[100], int arr2[100] , int  arrLength , int arr2Length)
{

    for (int i = 0; i < arrLength; i++)
    {
        if (!IsNumberInArray(arr[i], arr2, arrLength))
        {

            AddArrayElement(arr[i], arr2, arr2Length);

        }
    };
}
