int CounterOfLetterCase( string S1 , char letter)
{
    short Counter = 0 ;

    for (short i = 0; i < S1.length(); i++)
    {
        if (S1[i]== letter)
        {
            Counter++;
        }
       
    }
    return Counter;
}
