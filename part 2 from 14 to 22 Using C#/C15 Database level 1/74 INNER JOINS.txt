

 -- Join between 2 tables

SELECT Employees.ID, Employees.FirstName, Employees.LastName, Departments.Name as DepName
FROM     Departments INNER JOIN Employees 
ON Departments.ID = Employees.DepartmentID
				  Where Departments.Name = 'IT';
				  
 -- Join between 3 tables

 
SELECT Employees.ID, Employees.FirstName, Employees.LastName, Departments.Name, Countries.Name AS Expr1
FROM     Employees INNER JOIN
                  Departments ON Employees.DepartmentID = Departments.ID INNER JOIN
                  Countries ON Employees.CountryID = Countries.ID;
