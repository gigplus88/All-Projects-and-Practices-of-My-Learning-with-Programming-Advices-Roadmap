using System;
using System.Runtime.InteropServices;

class Program
{
    // Import MessageBox function from user32.dll
    [DllImport("user32.dll", CharSet = CharSet.Unicode, SetLastError = true)]
    static extern int MessageBox(IntPtr hWnd, String text, String caption, int type);

    static void Main()
    {
        MessageBox(IntPtr.Zero, "Hello, World!", "My Message Box from OS", 0);
        MessageBox(IntPtr.Zero, "Hello, World!", "My Message Box from OS", 1);
        MessageBox(IntPtr.Zero, "Hello, World!", "My Message Box from OS", 3);

        Console.WriteLine( "Message boxes appear sucssesffully");
    }
}
