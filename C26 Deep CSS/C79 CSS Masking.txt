    .demo1 {

    border-radius: 16px;        /* Round corners; the mask will intersect with this radius */

    /* üî≤ CSS Mask: radial ‚Äúspotlight‚Äù
      circle at 50% 50%   ‚Üí center the gradient in the box (X=50%, Y=50%)
      rgba(...,1) 45%     ‚Üí fully visible disk out to 45% of the box‚Äôs smaller dimension
      rgba(...,0) 60%     ‚Üí fade to fully transparent by 60% (soft edge/vignette)
      Anything beyond 60% ‚Üí fully masked (hidden)
    */
    mask-image: radial-gradient(
      circle at 50% 50%,
      rgba(0,0,0,1) 45%,
      rgba(0,0,0,0) 60%
    );

    -webkit-mask-image: radial-gradient(
      circle at 50% 50%,
      rgba(0,0,0,1) 45%,
      rgba(0,0,0,0) 60%
    );

    /* üö´ Don‚Äôt tile the mask image (default is repeat) */

    -webkit-mask-repeat: no-repeat;
