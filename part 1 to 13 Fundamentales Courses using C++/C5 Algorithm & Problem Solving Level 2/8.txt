bool CheckPerfect(int Number)
{
     int sum = 0;
    for (int i = 1; i <= Number - 1; i++)
    {

        if (Number % i == 0)
        {
            sum = sum + i;

        }

    }
    return Number == sum;
    
}

{

    }

}

int GetNumberOfDigit( int Number , int Digit)
{
   
    int Remainder = 0 , FreqCount =0 ;
  
    while (Number>0)
    {
        Remainder = Number % 10;
        if (Remainder == Digit)
        {
            FreqCount = FreqCount + 1;
        }
        
        Number = Number / 10;

    }
    return  FreqCount;
}
