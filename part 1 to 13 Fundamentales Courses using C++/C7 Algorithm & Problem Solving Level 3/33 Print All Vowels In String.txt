bool IsVowel(char letter)
{

    letter = tolower(letter);

    return((letter == 'a') || (letter == 'o') || (letter == 'u') || (letter == 'i') || (letter == 'e'));
}

void PrintVowelsInString(string  S1)
{
    for (short i = 0; i < S1.length(); i++)
    {
        if (IsVowel(S1[i]))
        {
            cout << S1[i] <<"\t";
        }
    }
   
}
