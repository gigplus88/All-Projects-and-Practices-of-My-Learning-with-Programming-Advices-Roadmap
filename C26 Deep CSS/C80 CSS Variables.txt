<button id="toggleBtn">Switch to Dark Mode</button>
<button id="toggleBtnRose">Switch to Rose Mode</button>


<script>
  const btn = document.getElementById('toggleBtn');
  const Rosebtn = document.getElementById('toggleBtnRose');

  const body = document.body;

  btn.addEventListener('click', () => {
    body.classList.toggle('dark');

    if (body.classList.contains('dark')) {
      btn.textContent = 'Switch to Light Mode';
    } else {
      btn.textContent = 'Switch to Dark Mode';
    }
  });


  Rosebtn.addEventListener('click', () => {
    body.classList.toggle('rose');

    if (body.classList.contains('rose')) {
      Rosebtn.textContent = 'Switch to Light Mode';
    } else {
      body.classList.toggle('light');
      Rosebtn.textContent = 'Switch to Rose Mode';
    }
  });
</script>

<h1>Resizable Layout with CSS Variables</h1>

<section>
  <p>This layout uses CSS variables for font sizes and spacing.</p>
  <p>Change them with JavaScript and see everything update instantly.</p>
</section>

<button id="increase">Increase Size</button>
<button id="decrease">Decrease Size</button>

<script>
  const root = document.documentElement;

  document.getElementById('increase').addEventListener('click', () => {
    // Get current values
    let mainFont = parseInt(getComputedStyle(root).getPropertyValue('--main-font-size'));
    let headingFont = parseInt(getComputedStyle(root).getPropertyValue('--heading-font-size'));
    let padding = parseInt(getComputedStyle(root).getPropertyValue('--section-padding'));

    // Increase each by 2px
    root.style.setProperty('--main-font-size', Math.min(mainFont + 2 , 25) + 'px');
    root.style.setProperty('--heading-font-size',Math.min(headingFont + 2 , 35)  + 'px');
    root.style.setProperty('--section-padding',Math.min(padding + 2 , 26) + 'px');
  });

  document.getElementById('decrease').addEventListener('click', () => {
    let mainFont = parseInt(getComputedStyle(root).getPropertyValue('--main-font-size'));
    let headingFont = parseInt(getComputedStyle(root).getPropertyValue('--heading-font-size'));
    let padding = parseInt(getComputedStyle(root).getPropertyValue('--section-padding'));

    // Decrease each by 2px
    root.style.setProperty('--main-font-size', Math.max(mainFont - 2, 10) + 'px');
    root.style.setProperty('--heading-font-size', Math.max(headingFont - 2, 14) + 'px');
    root.style.setProperty('--section-padding', Math.max(padding - 4, 8) + 'px');
  });
</script>
