int GetNumberOfDigit( int Number , int Digit)
{
   
    int Remainder = 0 , FreqCount =0 ;
  
    while (Number>0)
    {
        Remainder = Number % 10;
        if (Remainder == Digit)
        {
            FreqCount = FreqCount + 1;
        }
        
        Number = Number / 10;

    }
    return  FreqCount;
}

void PrintAllDigits(int Number)
{
    short  DigitFrequency = 0;
    
        for (short i = 0; i <=9; i++)
        {

            if (DigitFrequency >0)

                    << DigitFrequency << " Time(s)." << endl;

}
