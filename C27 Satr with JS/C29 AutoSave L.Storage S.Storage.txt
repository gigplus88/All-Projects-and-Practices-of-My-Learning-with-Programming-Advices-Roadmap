   <script>
   
   if (sessionStorage.getItem("autoSaveContent") === null) {
    sessionStorage.setItem("autoSaveContent", "");
  }

  const textSaved = sessionStorage.getItem("autoSaveContent");


  if (textSaved !== null) {
    document.getElementById("textbox2").value = textSaved;
  }

  const autoSave = document.getElementById("textbox2");


  autoSave.addEventListener("input", (e) => {
    const currentText = e.target.value; 
    sessionStorage.setItem("autoSaveContent", currentText);
  });





  /*With Local Storage*/
   if (localStorage.getItem("aSContentLS") === null) {
    localStorage.setItem("aSContentLS", "");
  }

  const textSavedLS = localStorage.getItem("aSContentLS");


  if (textSavedLS !== null) {
    document.getElementById("textbox3").value = textSavedLS;
  }

  const autoSaveLS = document.getElementById("textbox3");


  autoSaveLS.addEventListener("input", (e) => {
    const currentText = e.target.value; 
    localStorage.setItem("aSContentLS", currentText);
  });

</script>
