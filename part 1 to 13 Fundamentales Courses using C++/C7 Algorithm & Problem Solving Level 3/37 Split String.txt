#include <iostream>
#include <string>
#include <vector>
#include <iomanip>

using namespace std;


string ReadString()
{
    string  S1;
    cout << "Please Enter Your String?" << endl;
    getline(cin, S1);

    return S1;
}

vector<string> SplitString(string  S1 , string delim )
{
        vector<string> vSplit;
        string Word;
        short pos = 0 ;

    

        while ((pos = S1.find(delim)) != std::string::npos )
        {
            Word = S1.substr(0, pos);

            if (Word != "")
            {
                vSplit.push_back(Word)  ;
            
            }

            S1.erase(0, pos + delim.length());
        }

    

    if (S1 != "")
    {
        vSplit.push_back(S1);
    }

    return vSplit;


}





int main()
{
    vector<string> vSplit;

    vSplit = SplitString(ReadString(), " ");
    
    cout << "Tokens = " << vSplit.size() << endl;

    for (string word : vSplit)
    {
        cout << word << endl;

    }



    system("pause>0");

    return 0;
}
