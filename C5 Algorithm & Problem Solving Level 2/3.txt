bool IsPerfectNumber(int Number)
{
    int sum = 0;
    for (int i = 1; i <= Number - 1; i++)
    {

        if (Number % i == 0)
        {
            sum = sum + i;

        }

    }
    return Number == sum;

}

void PrintPerfectOrNo()
{
