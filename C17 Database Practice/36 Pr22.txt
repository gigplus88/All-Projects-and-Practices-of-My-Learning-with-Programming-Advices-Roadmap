
select (cast (TotalWithNoSpecifiedDoors as float) / cast(TotalOfVehicls as float)) * 100 as Perc  from
(
select distinct  count(*) as TotalWithNoSpecifiedDoors , (select count(*) from VehicleDetails) as TotalOfVehicls
from VehicleDetails
where NumDoors is Null )R1  ;
