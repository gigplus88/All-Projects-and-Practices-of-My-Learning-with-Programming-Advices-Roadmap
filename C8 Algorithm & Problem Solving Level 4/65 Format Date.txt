vector<string> SplitString(string  S1, string delim)
{
    vector<string> vSplit;
    string Word;
    short pos = 0;



    while ((pos = S1.find(delim)) != std::string::npos)
    {
        Word = S1.substr(0, pos);

        if (Word != "")
        {
            vSplit.push_back(Word);


        }

        S1.erase(0, pos + delim.length());
    }



    if (S1 != "")
    {
        vSplit.push_back(S1);
    }

    return vSplit;


}

string ReplaceWordsInString(string S1, string WordToReplace, string NewWord)
{
    short  pos = S1.find(WordToReplace);

    while (pos != std::string::npos)
    {
        S1 = S1.replace(pos, WordToReplace.length(), NewWord);
        pos = S1.find(WordToReplace); //   find the next

    }
    return S1;

}

string FormatingDate(sDate Date  , string DateFormat = "dd/mm/yyyy")
{
    string FormatedDateString = "";
    FormatedDateString = ReplaceWordsInString(DateFormat, "dd", to_string(Date.Day));
    FormatedDateString = ReplaceWordsInString(FormatedDateString, "mm", to_string(Date.Month));
    FormatedDateString = ReplaceWordsInString(FormatedDateString, "yyyy", to_string(Date.Year));

    return FormatedDateString;

}
