        <!--──────────────────────────────────────────────
      │ Code Overview
      │ Purpose        → Show all results (success + failure) without stopping
      │ Layout         → Single HTML report UI
      │ Use Cases      → Batch jobs, dashboards, partial failures
      │ Key Properties → Promise.allSettled(), status/value/reason
      └──────────────────────────────────────────────-->
        <h2 style="margin: 0 0 8px">allSettled Report</h2>
        <p class="muted" style="margin: 0 0 14px">
          Runs tasks. Some fail. We still show every result.
        </p>

        <button id="btn">Run Batch</button>
        <div class="grid" id="grid"></div>
      </div>
    </div>

    <script>
      const btn = document.getElementById("btn");
      const grid = document.getElementById("grid");

      //Reusability of Promise func with name , delay and status of this Promise
      function task(name, delayMs, fail = false) {
        return new Promise((resolve, reject) => {
          setTimeout(() => {
            if (fail) reject(new Error(`${name} failed`));
            else resolve(`${name} done in ${delayMs}ms`);
          }, delayMs);
        });
      }

      btn.addEventListener("click", async () => {
        btn.disabled = true;
        grid.innerHTML = "";

        // Run multiple tasks; we want *all* outcomes
        // The time of rendering page is the gretest Delay of Promises
        const results = await Promise.allSettled([
          task("Resize images", 600, false),
          task("Upload files", 1600, true),
          task("Send emails", 500, false),
          task("Write logs", 300, true),
        ]);

        // Render the report
        grid.innerHTML = results
          .map((r, i) => {
            if (r.status === "fulfilled") {
              return `
            <div class="box">
              <div class="ok" style="font-weight:800;">✅ Fulfilled</div>
              <div class="muted" style="margin-top:6px;">${r.value}</div>
            </div>
          `;
            }
            return `
          <div class="box">
            <div class="bad" style="font-weight:800;">❌ Rejected</div>
            <div class="muted" style="margin-top:6px;">${r.reason.message}</div>
          </div>
        `;
          })
          .join("");

        btn.disabled = false;
      });
    </script>
