  }

  /* 2) min() - height = smaller of 20vw or 150px */
  .min-box {
    background: #00aa55;
    height: min(20vw, 150px);
  }

  /* 3) max() - height = larger of 100px or 10vw */
  .max-box {
    background: #cc6600;
    height: max(100px, 10vw);
  }

  /* 4) clamp() - font-size between 16px and 32px, preferred 5vw */
  .clamp-box {
    background: #aa33aa;
