{
    bool Num;

    return (Year % 400 == 0) || ((Year % 100 != 0) && (Year % 4 == 0));


}

short MonthToDays(short Year, short Month)
{

    if (Month < 1 || Month> 12)
    {
        return 0;
    }

    short NumberOfDays[12] = { 31 , 28 , 31 , 30 , 31 ,30 , 31 , 31 , 30 , 31 , 30 , 31 };
    return  (Month == 2) ? IsLeapYear(Year) ? 29 : 28 : NumberOfDays[Month - 1];

}

sDate DecreaseDateByOneMonth(sDate Date)
{
    if (Date.Month == 1) // الصيغة العامة ثم سيعدل عليها فس السطر 163 و 164
    {
        Date.Month = 12;
        Date.Year--;
    }

    else
    {
        Date.Month--; // 31/2/2022
    }
    // We are two exception.
    // 1) for Month two , ex if we are 31 /1/2022 increasing one month should not be 31/2/2022 , it should be 28 /2/2022.
    // And other is 31/3/2022 to 31/4/2022 ىot exist.

    short NumberOfDaysInCurrentMonth = MonthToDays(Date.Year, Date.Month); //month=2 , year= 2022
    if (Date.Day > NumberOfDaysInCurrentMonth)
    {
        Date.Day = NumberOfDaysInCurrentMonth;
    }

    return Date;
}
