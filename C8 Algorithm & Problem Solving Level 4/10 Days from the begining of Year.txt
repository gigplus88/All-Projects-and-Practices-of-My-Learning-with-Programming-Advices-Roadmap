bool IsLeapYear(short Year)
{
    bool Num;

    return (Year % 400 == 0) || ((Year % 100 != 0) && (Year % 4 == 0));




}

short DayOrder(short Year , short Month , short Day)
{
    short a, y, m;
    a = (14 - Month) / 12;
    y = Year - a;
    m = Month + (12 * a) - 2;
    short d = (Day + y + (y / 4) - (y / 100) + (y / 400) + ((31 * m) / 12)) % 7;

    return d;

   
}

short MonthToDays(short Year, short Month)
{

    if (Month < 1 || Month> 12)
    {
        return 0;
    }

    short NumberOfDays[12] = { 31 , 28 , 31 , 30 , 31 ,30 , 31 , 31 , 31 , 30 , 31 , 30 };
    return  (Month == 2) ? IsLeapYear(Year) ? 29 : 28 : NumberOfDays[Month - 1];

}



short DayfromTheBeginingOfyear(short Year, short Month, short Day)
{
    short DayOrderFromFirstDay = 0;
    for (short i = 1; i <= Month-1; i++)
    {
        
        DayOrderFromFirstDay += MonthToDays( Year,  i);
    }

    return DayOrderFromFirstDay + Day;

}
