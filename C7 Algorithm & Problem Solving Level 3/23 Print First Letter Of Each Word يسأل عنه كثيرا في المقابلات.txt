{

    {
        if (YourInput[i] != ' ' && isFirstLetter)
        {
            cout << YourInput[i] << endl;
        }

        isFirstLetter = (YourInput[i] == ' '? true : false); //  لكي لا أكتب ما بعد هذا الحرف على أنه أول حرف في الكلمة فإذا وصل إلى الفراغ أعطى ترو فكتب مابعد الفراغ في المرة المقبلة  (true) كأني أقول له أني على غير الفراغ فلا تعطني 

    }
        
   
}
