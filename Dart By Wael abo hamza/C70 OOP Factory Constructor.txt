class People {
  final double lengh;
  final double breath;
  People.One(this.lengh, this.breath) {
    double Area = lengh * breath;
    print("Area is:{$Area}");
  }

// By factory i should use return its class or other class 
// By factory constructor i cant use the properties of class 
  factory People(lengh, breath) {
    if (lengh > 0 && breath > 0) {
      return People.One(lengh, lengh);
    } else {
      throw Exception("Must have more than 0");
    }
    
  }
}
