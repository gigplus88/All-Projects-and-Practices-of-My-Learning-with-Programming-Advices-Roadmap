
using System.Threading;
using System;

class Program
{
    static int sharedCounter = 0;
    static object lockObject = new object();


    static void Main()
    {
        // Create two threads that increment a shared counter
        Thread t1 = new Thread( ()=>IncrementCounter("Thread1") );
        Thread t2 = new Thread(() => IncrementCounter("Thread2")) ;


        t1.Start();
        t2.Start();


        // Wait for both threads to complete
        t1.Join();
        t2.Join();


        Console.WriteLine("Final Counter Value: " + sharedCounter);

        Console.ReadKey();
    }


    static void IncrementCounter(string ThreadName)
    {
        for (int i = 0; i < 100000; i++)
        {
            // Use lock to synchronize access to the shared counter
            lock (lockObject)
            {
                Console.WriteLine(ThreadName + ": " + sharedCounter);
                
                sharedCounter++;
                Thread.Sleep(1000);
            }
            
        }
    }
}
