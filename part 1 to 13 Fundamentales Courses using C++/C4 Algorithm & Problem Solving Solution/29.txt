enum enOddOrEven { Odd = 1, Even =2 };

int ReadNumber ()
{
    int Number;
   
    cout << "Please enter  your number" << endl;
    cin >> Number;
    return Number;

}

enOddOrEven ChekOddOeEven(int Number)
{
    
    if (Number % 2 ==0)
    {
        return enOddOrEven::Even;
    }

    else
    {
        return enOddOrEven::Odd;
    }
}

int SumEvenNumbersFromNto1_UssingWhile (int N)
{
    cout << "Sum Even numbers using While statement: " << endl;

    int sum=0,i = 1;
    while (i<=N)
    {

        if (ChekOddOeEven(i) == enOddOrEven::Even)
        {
            sum = sum + i;

        }
        i++;
    }

    return sum;
}

int SumEvenNumbersFromNto1_UssingDoWhile(int N)
{
    cout << "Sum Even numbers using While statement: " << endl;

    int i = 1 , sum=0;
    do
    {
        if (ChekOddOeEven(i) == enOddOrEven::Even)
        {
            sum = sum + i;

        }
        i++;

    } while (i<=N);

    return sum;

}

int SumEvenNumbersFrom1toN_UssingFor(int N)
{
    int  sum = 0;
    cout << "Sum Even numbers using For statement: " << endl;
    for (int i = 1; i <= N; i++)
    {

        if (ChekOddOeEven(i)== enOddOrEven::Even)
        {
            sum = sum + i;

        }

    }
    return sum ;

}
