<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Simple Animation</title>
</head>
<body>
    <canvas id="animationCanvas" width="500" height="500" style="border:1px solid #000000;">
    </canvas>
    <script>
        var canvas = document.getElementById('animationCanvas');
        var ctx = canvas.getContext('2d');
        var x = canvas.width / 2;
        var y = canvas.height - 30;
        var dx = 2;
        var dy = -2;
        var ballRadius = 10;

        function drawBall() {
            ctx.beginPath();
            ctx.arc(x, y, ballRadius, 0, Math.PI*2);
            ctx.fillStyle = "#0095DD";
            ctx.fill();
            ctx.closePath();
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height); // Clears the canvas
            drawBall();
            if(x + dx > canvas.width-ballRadius || x + dx < ballRadius) {
                dx = -dx; // Change direction
            }
            if(y + dy > canvas.height-ballRadius || y + dy < ballRadius) {
                dy = -dy; // Change direction
            }
            x += dx;
            y += dy;
            requestAnimationFrame(animate); // Recreate animation
        }

        animate(); // Start the animation
    </script>
</body>
</html>















<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Interactive Drawing</title>
</head>
<body>
    <canvas id="drawingCanvas" width="600" height="400" style="border:1px solid #000000;">
    </canvas>
    <script>
        var canvas = document.getElementById('drawingCanvas');
        var ctx = canvas.getContext('2d');
        var painting = false;

        function startPosition(e) {
            painting = true;
            draw(e);
        }

        function finishedPosition() {
            painting = false;
            ctx.beginPath(); // Begin a new path to start a new line
        }

        function draw(e) {
            if (!painting) return;
            ctx.lineWidth = 5;
            ctx.lineCap = 'round';
            ctx.lineTo(e.clientX - canvas.offsetLeft, e.clientY - canvas.offsetTop);
            ctx.stroke();
            ctx.beginPath(); // Begin a new path for the next segment
            ctx.moveTo(e.clientX - canvas.offsetLeft, e.clientY - canvas.offsetTop);
        }

        canvas.addEventListener('mousedown', startPosition);
        canvas.addEventListener('mouseup', finishedPosition);
        canvas.addEventListener('mousemove', draw);
    </script>
</body>
</html>
