// Inheritance concepts;
class clsEmployee : public clsPerson
{

    }

    {
        _Department = Department;

    }

    {
        cout << "Info:";

        cout << "Phone          :" << GetPhone() << endl;
        cout << "Salary         :" << _Salary << endl;

};
