vector<string> SplitString(string  S1, string delim)
{
    vector<string> vSplit;
    string Word;
    short pos = 0;



    while ((pos = S1.find(delim)) != std::string::npos)
    {
        Word = S1.substr(0, pos);

        if (Word != "")
        {
            vSplit.push_back(Word);


        }

        S1.erase(0, pos + delim.length());
    }



    if (S1 != "")
    {
        vSplit.push_back(S1);
    }

    return vSplit;


}

string ConvertRecordToLine(sDate Date, string delim)
{
    string DateString = "";

    DateString = DateString + to_string(Date.Day) + delim;
    DateString = DateString + to_string(Date.Month) + delim;
    DateString = DateString + to_string(Date.Year) + delim;


    return  DateString.substr(0, DateString.length() - delim.length());
}
