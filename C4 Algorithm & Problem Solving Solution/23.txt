float CircleAreaByTriangle(float a, float b , float c )
{
    const float PI = 3.141592653589793238;
    float p = (a + b + c) / 2;
    float Area = PI* pow((a * b * c),2) / pow( 4 * sqrt(p * (p - a) * (p - b) * (p - c)),2);
     return Area;
}
void PrintArea(float Area)
{
    cout << "The Circle Area is: " << Area << endl;

}
