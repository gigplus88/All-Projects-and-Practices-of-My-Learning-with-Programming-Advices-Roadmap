using System;
using System.Collections;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;



public class SimpleList<T> : IList<T>
{
    private List<T> _items = new List<T>();

    public T this[int index]
    {
        set => _items[index] = value;

        get =>  _items [index];
    }


    IEnumerator IEnumerable.GetEnumerator() => _items.GetEnumerator();

    public IEnumerator<T> GetEnumerator() => _items.GetEnumerator();

    public int Count => _items.Count;
    public bool IsReadOnly => false;
    public void Add( T item) => _items.Add(item);
    public void Clear() => _items.Clear();
    public bool Contains(T item) => _items.Contains(item);
    public void CopyTo( T[] newArray ,int arrayIndex) => _items.CopyTo( newArray ,arrayIndex);
    public void RemoveAt(int index) => _items.RemoveAt(index);
    public bool Remove(T item) => _items.Remove(item);
    public int IndexOf(T item) => _items.IndexOf(item);
    public void Insert(int index , T item) => _items.Insert(index, item);

}
class Program
{

    static void Main()
    {
        SimpleList<string> myList = new SimpleList<string>();
        myList.Add("First");
        myList.Add("Second");
        myList.Add("Third");
        myList.Add("fourth");
        myList.Add("fifth");

        myList.Insert(1, "Inserted");

        if (myList.Contains("Second"))
        {
            myList.Remove("Second");
            Console.WriteLine("Second is Removed");
        }
        myList.RemoveAt(1);

        Console.WriteLine($"The index of Third is {myList.IndexOf("Third")}");

        Console.WriteLine("#######################");

        for (int i = 0; i < myList.Count; i++)
        {
            Console.WriteLine(myList[i]);

        }
        Console.ReadKey();

    }

 

}
