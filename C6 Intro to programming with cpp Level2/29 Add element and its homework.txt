#include <iostream>
#include <cmath>
#include <string>
#include <iomanip>
#include <vector>

using namespace std;

void ReadNumbers(vector <int>& vNumbers)
{
     
    int Number;

    char UserAnswer;
    do
    {
        cout << "Please enter your number: ";
        cin >> Number;

        vNumbers.push_back(Number);

        cout <<"Do you want add more numbers Y/N: ";
        cin >> UserAnswer;

    } while (UserAnswer == 'Y' || UserAnswer == 'y');

}

void PrintNumbers(vector <int> &vNumbers)
{
    cout << "Numbers Vectors: ";

    for (int& Number : vNumbers)
    {
        cout << Number << " ";
    }


}

int main()
{
    vector <int> vNumbers ;

    ReadNumbers(vNumbers);
    PrintNumbers(vNumbers);
    
   
    return 0;
}
