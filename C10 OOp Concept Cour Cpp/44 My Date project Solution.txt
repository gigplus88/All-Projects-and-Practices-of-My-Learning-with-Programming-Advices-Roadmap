	vector<string> SplitString(string  S1, string delim)
	{
		vector<string> vSplit;
		string Word;
		short pos = 0;



		while ((pos = S1.find(delim)) != std::string::npos)
		{
			Word = S1.substr(0, pos);

			if (Word != "")
			{
				vSplit.push_back(Word);


			}

			S1.erase(0, pos + delim.length());
		}



		if (S1 != "")
		{
			vSplit.push_back(S1);
		}

		return vSplit;


	}

	{

		if (Month < 1 || Month> 12)
		{
			return 0;
		}

		short NumberOfDays[12] = { 31 , 28 , 31 , 30 , 31 ,30 , 31 , 31 , 31 , 30 , 31 , 30 };
		return  (Month == 2) ? IsLeapYear(Year) ? 29 : 28 : NumberOfDays[Month - 1];

	}

	void GetDateFromDayOrderInYear(short DateOrderInYear, short Year)
	{
		short RemainingDays = DateOrderInYear;
		short MonthDays = 0;

		_Year = Year;
		_Month = 1;

		while (true)
		{
			MonthDays = MonthToDays(Year, _Month);
			if (RemainingDays > MonthDays)
			{
				RemainingDays -= MonthDays;
				_Month++;
			}

			else
			{
				_Day = RemainingDays;
				break;
			}
		}

	}

	// DayOrder

	static short DayOrder(short Year, short Month, short Day)
	{
		short a, y, m;
		a = (14 - Month) / 12;
		y = Year - a;
		m = Month + (12 * a) - 2;
		short d = (Day + y + (y / 4) - (y / 100) + (y / 400) + ((31 * m) / 12)) % 7;

		return d;

	}

	static short DayOrder(clsDate Date)
	{
		return DayOrder(Date.Year, Date.Month, Date.Day);
	}

		return DayOrder(_Year, _Month, _Day);

	// Day From The Begining Of Year

	static short DayFromTheBeginingOfYear(short Year, short Month, short Day)
	{
		short DayOrderFromFirstDay = 0;
		for (short i = 1; i <= Month - 1; i++)
		{

			DayOrderFromFirstDay += MonthTodays(Year, i);
		}

		return DayOrderFromFirstDay + Day;

	}

	{
		return DayFromTheBeginingOfYear(_Year, _Month, _Day);
	}

	// Date From Day Order In Year

	static clsDate DateFromDayOrderInYear(short DateOrderInYear, short Year)
	{
		clsDate Date;
		short RemainingDays = DateOrderInYear;
		short MonthDays = 0;

		Date.Year = Year;
		Date.Month = 1;

		while (true)
		{
			MonthDays = MonthTodays(Year, Date.Month);
			if (RemainingDays > MonthDays)
			{
				RemainingDays -= MonthDays;
				Date.Month++;
			}

			else
			{
				Date.Day = RemainingDays;
				break;
			}
		}

		return Date;
	}

			}

		}

	}

	// Increase Day By One Day

	static clsDate IncreaseDateByOneDay(clsDate Date)
	{
		if (CheckIsLastDayInMonth(Date))
		{
			if (CheckIsLastMonthInYear(Date))
			{
				Date.Month = 1;
				Date.Day = 1;
				Date.Year++;
			}

			else
			{
				Date.Day = 1;
				Date.Month++;
			}
		}

		else
		{
			Date.Day++;
		}
		return Date;

	}

	// Difference Days

	static int DifferenceDays(clsDate Date1, clsDate Date2, bool includeEndDay = false)
	{
		int Day = 0;

		while (IsDate1BeforeDate2(Date1, Date2))
		{
			Day++;
			Date2 = IncreaseDateByOneDay(Date2);
		}

		return includeEndDay ? ++Day : Day;

	}

	int DifferenceDays(clsDate Date2, bool includeEndDay = false)
	{
		return DifferenceDays(*this, Date2, false); 
	}

	static short CalculateBusinessDays(clsDate DateFrom, clsDate DateTo)
	{

		short Days = 0;
		while (IsDate1BeforeDate2(DateFrom, DateTo))
		{
			if (IsItBusinessDay(DateFrom))
				Days++;

			DateFrom = IncreaseDateByOneDay(DateFrom);
		}

		return Days;

	}

			if (IsItWeekend(DateFrom))
				WeekEndCounter++;

			DateFrom = IncreaseDateByOneDay(DateFrom);
		}
		//to add weekends 
		for (short i = 1; i <= WeekEndCounter; i++)
			DateFrom = IncreaseDateByOneDay(DateFrom);

		return DateFrom;
	}
	
	// Date 1 after Date 2

	static bool IsDate1AfterDate2(clsDate Date1, clsDate Date2)
	{
		return !IsDate1BeforeDate2(Date2, Date1) && !DatesAreEqualDate(Date1, Date2);

	}

	bool IsDate1AfterDate2( clsDate Date2)
	{
		return IsDate1AfterDate2(*this, Date2);
	}

	// Compare Dates

	enum enCompareDate { Date1Before = -1, Date1EqualDate2 = 0, Date1After = 1 };

	static enCompareDate ComparDates(clsDate Date1, clsDate Date2) // Enum can give me number results;
	{
		if (IsDate1BeforeDate2(Date2, Date1))
		{
			return enCompareDate::Date1Before;
		}

		else if (DatesAreEqualDate( Date1,  Date2))
		{
			return enCompareDate::Date1EqualDate2;

		}

		return enCompareDate::Date1After;

	}

	enCompareDate ComparDates( clsDate Date2) // Enum can give me number results;
	{
		return ComparDates(*this, Date2);
	}
