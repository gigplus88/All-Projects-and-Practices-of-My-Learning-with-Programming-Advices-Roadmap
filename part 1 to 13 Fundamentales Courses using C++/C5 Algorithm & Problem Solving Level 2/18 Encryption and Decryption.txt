string EncryptText(string Text, short  EncryptionKey)
{
    for (int i = 0; i <= Text.length() ; i++)
    {
        Text[i] = char( (int)Text[i] + EncryptionKey);

    }
   
    return Text;
    
}

string DecryptText(string Text, short  EncryptionKey)
{
    for (int i = 0; i <= Text.length(); i++)
    {
        Text[i] = char((int)Text[i] - EncryptionKey);

    }

    return Text;

}
