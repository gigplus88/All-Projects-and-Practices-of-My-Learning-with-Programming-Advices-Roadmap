    h2 { margin-top: 2rem; font-size: 1.2rem; color: #333; }

    .grid {
      display: grid;
      grid-template-columns: repeat(3, 100px);
      grid-template-rows:    repeat(3, 80px);
      gap: 10px;
      margin-top: 0.5rem;
      background: #eee;
      padding: 10px;
    }
    .grid div {
      background: #4a90e2;
      color: white;
      display: flex;
      align-items: center;
      justify-content: center;
      font-weight: bold;
    }

    /* 1) Default row flow */
    .flow-row {
      grid-auto-flow: row; /* items: 1→3 fill row 1, 4→6 wrap to row 2 */
    }

    /* 2) Column flow */
    .flow-column {
      grid-auto-flow: column; /* items: 1→2 fill col 1, 3→4 wrap to col 2, etc. */
    }

    /* 3) Row dense */
    .flow-row-dense {
      grid-auto-flow: row dense;
      /* intentionally leave a gap by spanning item3 */
    }
    /* force item3 to span two columns to create a hole */
    .flow-row-dense .item3 { grid-column: span 3; }

    /* 4) Column dense */
    .flow-col-dense {
      grid-auto-flow: column dense;
    }
    .flow-col-dense .item3 { grid-row: span 3; } /* hole in column 1, row 2 */
  </style>
</head>
