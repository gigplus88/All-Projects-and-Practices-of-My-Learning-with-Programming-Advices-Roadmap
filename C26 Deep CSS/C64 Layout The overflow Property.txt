      font-size: 16px;
    }
    /* No Truncation (shows all text, may wrap) */
    .no-truncation {
      white-space:normal;
      overflow: visible;
      background: #eaf6fb;
    }
    /* Single-Line Ellipsis */
    .single-line-ellipsis {
      white-space:nowrap;
      text-overflow:ellipsis;
      overflow: hidden;
      background: #fde9e9;
    }
    /* Multi-Line Clamp: Show up to 3 lines, then ellipsis */
    .multi-line-clamp {
      display: -webkit-box;
      -webkit-line-clamp: 3;
      -webkit-box-orient:vertical ;
      overflow: hidden;
      text-overflow: ellipsis;
      background: #e9fbe9;
      height: 3.9em; /* 3 lines x line-height (1.3em) */
      line-height: 1.3em;
    }
    /* Clipped (hidden overflow, no scroll, no ellipsis) */
    .clipped {
      overflow: hidden;
      white-space: nowrap;
      background: #ece9fa;
    }
    /* Tooltip Example */
    .tooltip {
      overflow: hidden;
      white-space: nowrap;
      text-overflow: ellipsis;
      background: #faf6e9;
      cursor: pointer;
    }
  </style>
